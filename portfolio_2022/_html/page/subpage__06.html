$(document).ready(function(){
  let move = 0;
  //게시물이 이동되는 값을 초기화!
  const elements = $(".item_wrap > span");
  const moveAdd = (window.innerWidth / 3);
  //디바이스 너비 / 3
  const length = $(elements).length;
  //총 게시물 개수
  $(".btn_circle .prev").bind("click",function(){
  //prev 버튼을 클릭하면 아래의 이벤트가 발생된다.
  	if($(this).hasClass("disable"))return false;
  	//disable이 있다면 여기에서 멈춘다(이전 게시물이 없으므로 이벤트가 진행되지 않는다).
  	$(".btn_circle .next").removeClass("disable"); 
  	//next버튼을 클릭하면 현재 prev버튼에서 disable제거(이벤트 시작)
  	$(elements).filter(".on").removeClass("on").prev("span").addClass("on");
  	//현재 게시물에서 .on을 제거하고, 다음 게시물에 .on을 추가한다.
  	if($(elements).index($(elements).filter(".on")) == 0){
  	//이전 게시물이 없을 때 
  		$(".btn_circle .prev").addClass("disable");
  		//disable 추가
  	}
  	move += moveAdd;
  	//버튼을 누를 때마다 1칸의 너비만큼 값이 더해진다.
  	$(".item_wrap").css("transform","translateX("+move+"px)");
  	//버튼을 누를 때마다 transform을 이용해 1칸의 너비만큼 이동된다.
  });
  
  $(".btn_circle .next").bind("click",function(){
  	if($(this).hasClass("disable"))return false;
  	$(".btn_circle .prev").removeClass("disable");
  	$(elements).filter(".on").removeClass("on").next("span").addClass("on");
  	if($(elements).index($(elements).filter(".on")) == (length-2)){
  		$(".btn_circle .next").addClass("disable");
  	} 
  	move -= moveAdd;
  	$(".item_wrap").css("transform","translateX("+move+"px)");
  });
})

디바이스 크기에 따라 3칸으로 표출되고, 버튼을 누를때마다 1칸의 크기만큼 이동이 가능하다. 게시물이 없을 시 button에 .disable이 추가되어 이동이 불가능하다.
