const accordion = document.querySelectorAll(".accordion");
//마크업페이지에서 .accordion을 모두 찾아 accordion이라 칭한다.
  for (let i = 0; i < accordion.length; i++) {
//i가 0부터 시작되고, accordion의 길이보다 작을 때 1씩 증가
    accordion[i].addEventListener("click", function() {
//accordion을 클릭할 때마다 아래의 이벤트가 시작된다.
    this.classList.toggle("active");
//현재 클릭한 accordion에 클래스 active 추가
    const panel = this.nextElementSibling;
//현재 클릭한 accordion의 바로 다음 요소(여기에서는 div)를 가져와 panel이라 칭한다. 
    if (panel.style.maxHeight) {
      panel.style.maxHeight = '';
//panel의 높이값이 있을 경우, 높이값을 삭제한다.
    } else { 
      panel.style.maxHeight = panel.scrollHeight + "px";
//panel의 높이값이 없을 경우, panel에 들어있는 컨텐츠의 전체 높이(패딩과 테두리 포함, 마진 제외)를 px로 입력한다.
  }
});
}
