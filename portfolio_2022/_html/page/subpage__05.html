const clockTitle = document.querySelector(".day");
//day 클래스를 찾아서 clockTitle이라는 변수에 넣는다
function clockTimer() {
  const cDate = new Date();
  //날짜 찾기
  const dDay = new Date("april 25, 2023");
  //기준 날짜를 dDay에 넣는다.
  const reDay = (dDay - cDate) / 1000; 
  //기준 날짜 - 현재날짜
  밀리초 단위의 숫자 값으로 반환(312,000 day)하므로 1000 나누기
  const oneHour = 60 * 60; 
  //1시간을 초로 환산
  const calDay = String(Math.floor(reDay / (oneHour * 24))).padStart(2, "0");
  //구해진 디데이 값을 Math.floor를 사용하여 소수값을 버리고 정수만 가져온다. 
  padStart를 이용해 1~9까지를 01~09 두자리로 로 표출한다.
  clockTitle.innerText = `${calDay}`;
  //.day에 디데이 값 넣기
}
clockTimer();

let start = 1;
//시작하는 숫자 = 1
const day = clockTitle.innerText;
//clockTimer로 구한 값을 day에 넣는다.
const tId = setInterval(countUp, 15);
//countUp 호출, 0.015초 
function countUp() {
  start = start + 1;
  //1부터 원하는 시간(day)까지 1씩 증가
  if (start == day) {
    clearInterval(tId);
    //원하는 시간(day)에 도달하면 타이머가 멈춘다.
  }
  clockTitle.innerText = start
  //1부터 원하는 시간(day)까지 휘리링 효과 시작
}
