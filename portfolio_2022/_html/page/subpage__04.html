const imageArr = document.querySelectorAll(".gallery_list img")
//사진을 모두 가져온다.
const titleArr = document.querySelectorAll(".gallery_list p strong");
//제목을 모두 가져온다.
const layerViewDiv = document.getElementById("layerView");
const viewImgTag = document.getElementById("viewImg");
const viewTitle = document.getElementById("layerTitle");
let num = 1;
function viewLayer(num) {
//해당 목록을 누르면 이벤트가 시작된다.
  viewTitle.innerHTML = titleArr[num].innerText;
  //num번째 제목을 모달의 제목에 붙여 넣는다.
  viewImgTag.src = imageArr[num].src;
  //num번째 사진을 모달의 사진에 붙여 넣는다.
  viewImgTag.alt = titleArr[num].innerText;
  //num번째 제목을 모달창에 대체텍스트로  붙여 넣는다.
  layerViewDiv.className = layerViewDiv.className.replace("hide", "");
  //모달에 hide 클래스를 추가하여 숨김 처리 한다.
}
function changeImg(plusNum) {
  //이전&다음 버튼을 누르면 실행된다
  num = parseInt(num + plusNum);
  //현재 num과 방금 실행된 버튼의 plusNum를 더하여 num에 숫자로 변환하여 넣는다
  if (num >= imageArr.length) num = 0;
  //num이 사진의 개수와 같거나 더 크다면 num을 0으로 초기화 한다.
  viewTitle.innerHTML = titleArr[num].innerText;
  //num번째 제목을 모달의 제목에 붙여 넣는다.
  viewImgTag.src = imageArr[num].src;
  //num번째 제목을 사진에 붙여 넣는다.
  viewImgTag.alt = titleArr[num].innerText;
  //num번째 제목을 대체텍스트로 붙여 넣는다.
}
function closeViewLayer() {layerViewDiv.className += " hide";}
  //닫기 버튼을 누르면 모달에 hide 클래스를 추가하여 숨김 처리 한다.
